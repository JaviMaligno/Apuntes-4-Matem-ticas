\begin{ejercicio}{1} Sea $\x_1,\dotsc,\x_n$ una muestra aleatoria, entonces
$$
\sum_{i=1}^n (\x_i-\X) = 0
$$
\end{ejercicio}
\begin{solucion}
Basta tener en cuenta que 
$$
\sum_{i=1}^n (\x_i-\X) = \sum_{i=1}^n \x_i - \sum_{i=1}^n \X = \sum_{i=1}^n \x_i  - n \X = \sum_{i=1}^n \x_i  - n \frac{1}{n} \sum_{i=1}^n \x_i  =0
$$
\end{solucion}
