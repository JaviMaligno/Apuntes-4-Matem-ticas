\begin{ejercicio}{5}\label{1}
Sean $f\in\mathcal{F}(M,N)$, $p\in M$ y $g\in\mathcal{F}(f(p))$. Probar que $$f^*_p((dg)_{f(p)})=(d(g\circ f))_p.$$
\end{ejercicio}
\begin{solucion}
Sea $v\in T_p(M)$. 
$$f^*_p((dg)_{f(p)})v=(dg)_{f(p)}(f_{*p}v)=(f_{*p}v)g=v(g\circ f)=(d(g\circ f))_pv$$
\end{solucion}
